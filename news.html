    本站目録

    MyWeb
    ├── css # 存放樣式文件
    │ └── APlayer.min.css # 播放器插件樣式
    │ └── demo.css # 計算器樣式
    │ └── style.css # 本站主布局樣式
    ├── font # 存放字體文件
    │ └── misaki_gothic.ttf # 美咲黒體
    │ └── Pixel Millennium.ttf # 像素體
    ├── img # 存放圖片素材文件
    ├── js # 存放js脚本文件
    │ └── APlayer.min.js # 播放器插件
    │ └── code.js # 本站主布局視覺效果
    │ └── demo.js # 計算器主程序
    │ └── jquery-3.3.1.js # jquery插件
    │ └── typed.min.js # 實現打字效果插件
    ├── media # 存放音樂及封面
    ├── news.html # 開發文本頁面
    ├── about.html # 個人介紹頁面
    ├── calculator.html # 計算器頁面
    ├── gallery.html # 畫廊頁面
    └── index.html # 本站主入口頁面
